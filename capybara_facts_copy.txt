# Set the original file name and the output file name
original_file = "capybara_facts.txt"  # Replace with the actual input file name
output_file = original_file.replace(".txt", "_out.csv")  # Modify to _out.csv format

try:
    # Open the original file and read its content
    with open(original_file, 'r') as file:
        content = file.read()
    
    # Write the content to a new output file
    with open(output_file, 'w') as file:
        file.write(content)
    
    print(f"The file has been successfully copied as {output_file}")
except FileNotFoundError:
    print(f"The file {original_file} was not found.")
except Exception as e:
    print(f"An error occurred: {e}")
